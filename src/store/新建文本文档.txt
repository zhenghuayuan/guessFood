//引入vue及vuex
import Vue from 'vue'
import Vuex from 'vuex'
Vue.use(Vuex)
const state = {
    notes:[1],
    domain: "localhost:3000"
}
const mutations = {
	FN(){
		state.notes.push(1)
	},
}
const actions = {
	addNote({commit}){
		commit("FN")
	},
}
const getters = {
  notes: state => state.notes,
  domain: state => state.domain
}
export default new Vuex.Store({
    state,
    mutations,
    actions,
    getters
})
